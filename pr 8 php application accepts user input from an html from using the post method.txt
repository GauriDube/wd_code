    <?php

    if($_SERVER["REQUEST_METHOD"] == "POST")
    {
        $name = $_POST['name'];

        echo "<h2>Submitted Information</h2>";

        echo "Name: " . $name . "<br>";
        echo "Email: " . $email . "<br>";
        echo "Course: " . $course . "<br>";
    }
